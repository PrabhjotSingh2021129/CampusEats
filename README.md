  FRONTEND_URL : "",

  # To Start the Backend : 

  ``` 
  npm run dev
  ```

  # Frontend Variables : 
  
  REACT_APP_SERVER_DOMAIN = <backend url>
  REACT_APP_ADMIN_PANEL = <admin email id>
  REACT_APP_STRIPE_PUBLIC_KEY = <stripe public key>

  # To Start the Frontend : 
  ```
  npm start
  ```
